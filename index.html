import tkinter as tk import random import math # 文本内容（可以自己改） texts = [ "我想你了", "爱你", "好好吃饭", "早点睡", "保持开心", "天天开心", "顺顺利利", "我在", "别熬夜", "爱你哦" ] root = tk.Tk() root.title("❤️") root.attributes("-fullscreen", True) root.configure(bg="black") width = root.winfo_screenwidth() height = root.winfo_screenheight() labels = [] # ❤️ 生成爱心坐标（直接成型） def heart_points(scale=12): points = [] for t in range(0, 360, 5): t = math.radians(t) x = 16 * math.sin(t)**3 y = -(13 * math.cos(t) - 5 * math.cos(2*t) - 2 * math.cos(3*t) - math.cos(4*t)) points.append((x * scale + width/2, y * scale + height/2)) return points # 🎯 创建爱心（直接出现） def create_heart(): pts = heart_points() for (x, y) in pts: text = random.choice(texts) label = tk.Label(root, text=text, fg=random.choice(["#ff4d6d","#ff66cc","#66ffcc","#ffff66","#66ccff"]), bg="black", font=("微软雅黑", 12)) label.place(x=x, y=y) labels.append(label) # 💥 爱心直接消失 def remove_heart(): for lbl in labels: lbl.destroy() labels.clear() # 🎲 随机飘字 def random_text(): text = random.choice(texts) label = tk.Label(root, text=text, fg=random.choice(["#ff4d6d","#ff66cc","#66ffcc","#ffff66","#66ccff"]), bg="black", font=("微软雅黑", random.randint(10,18))) x = random.randint(0, width) y = random.randint(0, height) label.place(x=x, y=y) # 自动消失 root.after(2000, label.destroy) # 持续生成 root.after(150, random_text) # 执行流程 create_heart() # 直接生成爱心 root.after(3000, remove_heart) # 3秒后消失 root.after(3200, random_text) # 开始随机飘字 root.mainloop()
